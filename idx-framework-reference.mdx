### Project Management Commands

#### `sim init`

Initialize a new IDX application with boilerplate code.

**Syntax:**
```bash
sim init [OPTIONS]
```

**Options:**
- `--template <TEMPLATE>`: Use a specific template
- `--name <NAME>`: Set project name
- `--description <DESC>`: Set project description

**Available Templates:**
- `sample` (default): Basic Uniswap V3 pool indexer
- `contract-decoder`: Advanced contract interaction decoder
- `defi-tracker`: DeFi protocol activity tracker
- `nft-marketplace`: NFT marketplace event indexer
- `cross-chain`: Multi-chain data aggregator

**Examples:**
```bash
# Default initialization
mkdir my-indexer && cd my-indexer
sim init

# With specific template
sim init --template=contract-decoder --name="My Contract Indexer"

# Custom configuration
sim init --template=defi-tracker \
         --name="DeFi Analytics" \
         --description="Track DeFi protocol metrics"
```

**Generated Structure:**
```
my-indexer/
├── sim.toml              # Project configuration
├── listeners/            # Solidity listener contracts
│   ├── Listener.sol      # Main listener contract
│   └── test/            # Test contracts
├── apis/                # API implementation
│   ├── src/             # TypeScript source
│   ├── package.json     # Dependencies
│   └── wrangler.toml    # Cloudflare Workers config
├── .gitignore
└── README.md
```

#### `sim build`

Compile Solidity contracts using Foundry.

**Syntax:**
```bash
sim build [OPTIONS]
```

**Options:**
- `--watch`: Watch for file changes and rebuild
- `--verbose`: Show detailed compilation output
- `--clean`: Clean build cache before building

**Examples:**
```bash
# Standard build
sim build

# Watch mode for development
sim build --watch

# Clean build
sim build --clean --verbose
```

#### `sim test`

Run Solidity tests for listener contracts.

**Syntax:**
```bash
sim test [OPTIONS] [PATTERN]
```

**Options:**
- `--verbose`: Show detailed test output
- `--gas-report`: Generate gas usage report
- `--coverage`: Generate code coverage report
- `--fork-url <URL>`: Fork from specific blockchain URL
- `--fork-block <BLOCK>`: Fork from specific block number

**Examples:**
```bash
# Run all tests
sim test

# Run specific test pattern
sim test "test_pool_creation"

# Run with gas reporting
sim test --gas-report --verbose

# Fork testing from mainnet
sim test --fork-url https://eth-mainnet.alchemyapi.io/v2/YOUR-KEY \
         --fork-block 18500000
```

### Listener Management

#### `sim listeners evaluate`

Test listener contracts against real blockchain data.

**Syntax:**
```bash
sim listeners evaluate [OPTIONS]
```

**Required Options:**
- `--start-block <BLOCK>`: Starting block number
- `--chain-id <ID>`: Chain ID to evaluate against

**Optional Options:**
- `--end-block <BLOCK>`: Ending block number (default: start-block)
- `--limit <COUNT>`: Maximum number of blocks to process
- `--listener <NAME>`: Specific listener to evaluate
- `--verbose`: Show detailed processing information
- `--dry-run`: Simulate without persisting data

**Examples:**
```bash
# Evaluate single block
sim listeners evaluate --start-block 18500000 --chain-id 1

# Evaluate block range
sim listeners evaluate \
  --start-block 18500000 \
  --end-block 18500100 \
  --chain-id 1

# Evaluate specific listener
sim listeners evaluate \
  --start-block 18500000 \
  --chain-id 1 \
  --listener PoolCreationListener \
  --verbose

# Dry run evaluation
sim listeners evaluate \
  --start-block 18500000 \
  --chain-id 1 \
  --dry-run
```

### Listener Contract Structure

#### Basic Listener Template

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "sim-idx/Listener.sol";

contract MyListener is Listener {
    // Events that will create database tables
    event DataCaptured(
        address indexed contract_address,
        uint256 indexed block_number,
        uint256 timestamp,
        bytes32 data_hash
    );
    
    // Required: Process blockchain events
    function process(
        address contractAddress,
        bytes calldata data
    ) external override {
        // Decode and process event data
        emit DataCaptured(
            contractAddress,
            block.number,
            block.timestamp,
            keccak256(data)
        );
    }
    
    // Required: Filter which contracts to process
    function filter(address contractAddress) external view override returns (bool) {
        // Define your filtering logic
        return true; // Process all contracts
    }
}
```

#### Advanced Listener Example

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "sim-idx/Listener.sol";

contract DeFiProtocolListener is Listener {
    // Protocol-specific events
    event PoolCreated(
        address indexed token0,
        address indexed token1,
        uint24 indexed fee,
        address pool,
        uint256 block_number,
        uint256 timestamp
    );
    
    event SwapExecuted(
        address indexed pool,
        address indexed trader,
        address tokenIn,
        address tokenOut,
        uint256 amountIn,
        uint256 amountOut,
        uint256 block_number,
        uint256 timestamp
    );
    
    event LiquidityChanged(
        address indexed pool,
        address indexed provider,
        uint128 liquidity,
        bool isAdd,
        uint256 block_number,
        uint256 timestamp
    );
    
    // Supported protocol contracts
    mapping(address => bool) public supportedProtocols;
    mapping(bytes4 => string) public functionSelectors;
    
    constructor() {
        // Initialize supported protocols
        _initializeProtocols();
        _initializeFunctionSelectors();
    }
    
    function process(
        address contractAddress,
        bytes calldata data
    ) external override {
        if (!supportedProtocols[contractAddress]) {
            return;
        }
        
        // Extract function selector
        bytes4 selector = bytes4(data[:4]);
        
        // Route to appropriate handler
        if (selector == bytes4(keccak256("createPool(address,address,uint24)"))) {
            _processPoolCreation(contractAddress, data[4:]);
        } else if (selector == bytes4(keccak256("swap(address,bool,int256,uint160,bytes)"))) {
            _processSwap(contractAddress, data[4:]);
        } else if (selector == bytes4(keccak256("mint(address,int24,int24,uint128,bytes)"))) {
            _processLiquidityAdd(contractAddress, data[4:]);
        } else if (selector == bytes4(keccak256("burn(int24,int24,uint128)"))) {
            _processLiquidityRemove(contractAddress, data[4:]);
        }
    }
    
    function _processPoolCreation(address factory, bytes calldata data) internal {
        (address tokenA, address tokenB, uint24 fee) = 
            abi.decode(data, (address, address, uint24));
        
        // Calculate pool address (simplified)
        address pool = _calculatePoolAddress(tokenA, tokenB, fee);
        
        emit PoolCreated(
            tokenA,
            tokenB,
            fee,
            pool,
            block.number,
            block.timestamp
        );
    }
    
    function _processSwap(address pool, bytes calldata data) internal {
        (
            address recipient,
            bool zeroForOne,
            int256 amountSpecified,
            uint160 sqrtPriceLimitX96,
            bytes memory callbackData
        ) = abi.decode(data, (address, bool, int256, uint160, bytes));
        
        // Decode callback data to get actual trade details
        (address tokenIn, address tokenOut, uint256 amountIn, uint256 amountOut) = 
            _decodeSwapCallback(callbackData);
        
        emit SwapExecuted(
            pool,
            recipient,
            tokenIn,
            tokenOut,
            amountIn,
            amountOut,
            block.number,
            block.timestamp
        );
    }
    
    function _processLiquidityAdd(address pool, bytes calldata data) internal {
        (
            address recipient,
            int24 tickLower,
            int24 tickUpper,
            uint128 amount,
            bytes memory callbackData
        ) = abi.decode(data, (address, int24, int24, uint128, bytes));
        
        emit LiquidityChanged(
            pool,
            recipient,
            amount,
            true, // isAdd
            block.number,
            block.timestamp
        );
    }
    
    function _processLiquidityRemove(address pool, bytes calldata data) internal {
        (int24 tickLower, int24 tickUpper, uint128 amount) = 
            abi.decode(data, (int24, int24, uint128));
        
        emit LiquidityChanged(
            pool,
            msg.sender, // Assuming the caller is the liquidity provider
            amount,
            false, // isAdd
            block.number,
            block.timestamp
        );
    }
    
    function filter(address contractAddress) external view override returns (bool) {
        return supportedProtocols[contractAddress];
    }
    
    function _initializeProtocols() internal {
        // Uniswap V3
        supportedProtocols[0x1F98431c8aD98523631AE4a59f267346ea31F984] = true; // Factory
        supportedProtocols[0xE592427A0AEce92De3Edee1F18E0157C05861564] = true; // Router
        
        // Add more protocols as needed
    }
    
    function _initializeFunctionSelectors() internal {
        functionSelectors[bytes4(keccak256("createPool(address,address,uint24)"))] = "createPool";
        functionSelectors[bytes4(keccak256("swap(address,bool,int256,uint160,bytes)"))] = "swap";
        // Add more selectors as needed
    }
    
    function _calculatePoolAddress(address tokenA, address tokenB, uint24 fee) 
        internal 
        pure 
        returns (address) 
    {
        // Simplified pool address calculation
        return address(uint160(uint256(keccak256(abi.encodePacked(tokenA, tokenB, fee)))));
    }
    
    function _decodeSwapCallback(bytes memory data) 
        internal 
        pure 
        returns (address, address, uint256, uint256) 
    {
        // Decode swap callback data to extract trade details
        // Implementation depends on specific protocol callback format
        return (address(0), address(0), 0, 0);
    }
}
```

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "forge-std/Test.sol";
import "../Listener.sol";

contract ListenerTest is Test {
    DeFiProtocolListener public listener;
    
    // Test addresses
    address constant UNISWAP_V3_FACTORY = 0x1F98431c8aD98523631AE4a59f267346ea31F984;
    address constant USDC = 0xA0b86a33E6ba4D8a6c81Ca24c4F05d5c8f5b6F2e;

    function setUp() public {
        listener = new DeFiProtocolListener();
    }
    
    function testFilterValidProtocol() public {
        assertTrue(listener.filter(UNISWAP_V3_FACTORY));
    }
    
    function testFilterInvalidProtocol() public {
        assertFalse(listener.filter(address(0x1234)));
    }
    
    function testProcessPoolCreation() public {
        // Encode pool creation data
        bytes memory data = abi.encodeWithSignature(
            "createPool(address,address,uint24)",
            USDC,
            WETH,
            3000
        );
        
        // Expect PoolCreated event
        vm.expectEmit(true, true, true, false);
        emit PoolCreated(
            USDC,
            WETH,
            3000,
            address(0), // Calculated address
            block.number,
            block.timestamp
        );
        
        listener.process(UNISWAP_V3_FACTORY, data);
    }
    
    function testProcessSwap() public {
        bytes memory swapData = abi.encodeWithSignature(
            "swap(address,bool,int256,uint160,bytes)",
            address(this),
            true,
            1000000, // 1 USDC
            0,
            abi.encode(USDC, WETH, 1000000, 500000) // Mock callback data
        );
        
        vm.expectEmit(true, true, false, false);
        emit SwapExecuted(
            address(0x1234), // Mock pool
            address(this),
            USDC,
            WETH,
            1000000,
            500000,
            block.number,
            block.timestamp
        );
        
        listener.process(address(0x1234), swapData);
    }
    
    function testGasUsage() public {
        bytes memory data = abi.encodeWithSignature(
            "createPool(address,address,uint24)",
            USDC,
            WETH,
            3000
        );
        
        uint256 gasBefore = gasleft();
        listener.process(UNISWAP_V3_FACTORY, data);
        uint256 gasUsed = gasBefore - gasleft();
        
        // Ensure gas usage is reasonable
        assertLt(gasUsed, 100000, "Gas usage too high");
    }
    
    function testFuzzProcessing(address token0, address token1, uint24 fee) public {
        vm.assume(token0 != address(0) && token1 != address(0));
        vm.assume(token0 != token1);
        vm.assume(fee <= 1000000); // 100% max fee
        
        bytes memory data = abi.encodeWithSignature(
            "createPool(address,address,uint24)",
            token0,
            token1,
            fee
        );
        
        // Should not revert
        listener.process(UNISWAP_V3_FACTORY, data);
    }
}
```

### API Endpoints

#### Basic CRUD Operations

```typescript
// Get all pools
app.get('/api/pools', async (c) => {
  try {
    const limit = Math.min(parseInt(c.req.query('limit') || '100'), 1000);
    const offset = parseInt(c.req.query('offset') || '0');
    
    const pools = await db.client(c)
      .select()
      .from(poolCreated)
      .orderBy(desc(poolCreated.timestamp))
      .limit(limit)
      .offset(offset);
    
    return c.json({
      pools,
      pagination: {
        limit,
        offset,
        total: pools.length
      }
    });
  } catch (error) {
    console.error('Error fetching pools:', error);
    return c.json({ error: 'Failed to fetch pools' }, 500);
  }
});

// Get pool by address
app.get('/api/pools/:address', async (c) => {
  try {
    const address = c.req.param('address');
    
    const pool = await db.client(c)
      .select()
      .from(poolCreated)
      .where(eq(poolCreated.pool, address))
      .limit(1);
    
    if (pool.length === 0) {
      return c.json({ error: 'Pool not found' }, 404);
    }
    
    return c.json({ pool: pool[0] });
  } catch (error) {
    console.error('Error fetching pool:', error);
    return c.json({ error: 'Failed to fetch pool' }, 500);
  }
});

// Get swaps for a pool
app.get('/api/pools/:address/swaps', async (c) => {
  try {
    const address = c.req.param('address');
    const limit = Math.min(parseInt(c.req.query('limit') || '100'), 1000);
    const startTime = c.req.query('start_time');
    const endTime = c.req.query('end_time');
    
    let query = db.client(c)
      .select()
      .from(swapExecuted)
      .where(eq(swapExecuted.pool, address));
    
    // Add time filtering if provided
    if (startTime) {
      query = query.where(
        and(
          eq(swapExecuted.pool, address),
          gte(swapExecuted.timestamp, new Date(startTime))
        )
      );
    }
    
    if (endTime) {
      query = query.where(
        and(
          eq(swapExecuted.pool, address),
          lte(swapExecuted.timestamp, new Date(endTime))
        )
      );
    }
    
    const swaps = await query
      .orderBy(desc(swapExecuted.timestamp))
      .limit(limit);
    
    return c.json({ swaps });
  } catch (error) {
    console.error('Error fetching swaps:', error);
    return c.json({ error: 'Failed to fetch swaps' }, 500);
  }
});
```

```typescript
// Pool analytics
app.get('/api/analytics/pools', async (c) => {
  try {
    const timeframe = c.req.query('timeframe') || '24h';
    const startTime = getTimeAgo(timeframe);
    
    const client = db.client(c);
    
    // Pool creation metrics
    const poolMetrics = await client
      .select({
        totalPools: count(poolCreated.id),
        newPools: sql<number>`count(case when ${poolCreated.timestamp} >= ${startTime} then 1 end)`,
        avgPoolsPerDay: sql<number>`count(*) / extract(days from (now() - min(${poolCreated.timestamp})))`
      })
      .from(poolCreated);
    
    // Top pools by volume
    const topPools = await client
      .select({
        pool: swapExecuted.pool,
        swapCount: count(swapExecuted.id),
        totalVolumeIn: sql<string>`sum(${swapExecuted.amount_in})`,
        totalVolumeOut: sql<string>`sum(${swapExecuted.amount_out})`
      })
      .from(swapExecuted)
      .where(gte(swapExecuted.timestamp, startTime))
      .groupBy(swapExecuted.pool)
      .orderBy(desc(sql`sum(${swapExecuted.amount_in})`))
      .limit(10);
    
    // Trading activity over time
    const activityByHour = await client
      .select({
        hour: sql<string>`date_trunc('hour', ${swapExecuted.timestamp})`,
        swapCount: count(swapExecuted.id),
        volume: sql<string>`sum(${swapExecuted.amount_in})`
      })
      .from(swapExecuted)
      .where(gte(swapExecuted.timestamp, startTime))
      .groupBy(sql`date_trunc('hour', ${swapExecuted.timestamp})`)
      .orderBy(sql`date_trunc('hour', ${swapExecuted.timestamp})`);
    
    return c.json({
      timeframe,
      metrics: poolMetrics[0],
      topPools,
      activityByHour,
      generatedAt: new Date().toISOString()
    });
  } catch (error) {

    return c.json({ error: 'Failed to generate analytics' }, 500);
```

```typescript
// WebSocket endpoint for real-time updates
app.get('/api/ws/swaps', async (c) => {
  // Upgrade to WebSocket
  const upgrade = c.req.header('upgrade');
  if (upgrade !== 'websocket') {
    return c.text('Expected websocket', 426);
  }
  
  const { response, socket } = Deno.upgradeWebSocket(c.req.raw);
  
  socket.onopen = () => {
    console.log('WebSocket connection opened');

    socket.send(JSON.stringify({
      type: 'connected',
      timestamp: new Date().toISOString()
    }));
    
    // Start sending updates
    startRealtimeUpdates(socket);
  };
  
  socket.onmessage = (event) => {
    console.log('WebSocket message:', event.data);
  };
  
  socket.onclose = () => {
    console.log('WebSocket connection closed');
  };
  
  return response;
});

async function startRealtimeUpdates(socket: WebSocket) {
  let lastCheck = new Date();
  
  setInterval(async () => {
    try {
      // Get new swaps since last check
      const newSwaps = await db.client()
        .select()
        .from(swapExecuted)
        .where(gte(swapExecuted.timestamp, lastCheck))
        .orderBy(desc(swapExecuted.timestamp))
        .limit(50);
      
      if (newSwaps.length > 0) {
        socket.send(JSON.stringify({
          type: 'swaps',
          data: newSwaps,
          timestamp: new Date().toISOString()
        }));
        
        lastCheck = new Date(newSwaps[0].timestamp);
      }
    } catch (error) {
      console.error('Real-time update error:', error);
    }
  }, 5000); // Check every 5 seconds
}

            .select()
            .from(swapExecuted)
            .orderBy(desc(swapExecuted.timestamp))
            .limit(10);

          })}\n\n`;
          
          controller.enqueue(new TextEncoder().encode(eventData));
        } catch (error) {
          console.error('SSE error:', error);
        }
      }, 10000); // Update every 10 seconds
      
      // Clean up on close
      return () => {
        clearInterval(interval);
      };
    }
  });
  
  return new Response(stream, {

      'Content-Type': 'text/event-stream',
```

```typescript
// Time utility functions
function getTimeAgo(timeframe: string): Date {
  const now = new Date();
  
  switch (timeframe) {
    case '1h': return new Date(now.getTime() - 60 * 60 * 1000);
    case '24h': return new Date(now.getTime() - 24 * 60 * 60 * 1000);
    case '7d': return new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
    case '30d': return new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);
    default: return new Date(now.getTime() - 24 * 60 * 60 * 1000);
  }
}

// Validation middleware
const validateAddress = (address: string): boolean => {
  return /^0x[a-fA-F0-9]{40}$/.test(address);
};

app.use('/api/pools/:address/*', async (c, next) => {
  const address = c.req.param('address');
  if (!validateAddress(address)) {
    return c.json({ error: 'Invalid address format' }, 400);
  }
  await next();
});

// Error handling middleware
app.onError((err, c) => {
  console.error('API Error:', err);
  
  if (err.name === 'ValidationError') {
    return c.json({ error: err.message }, 400);
  }
  
  if (err.name === 'DatabaseError') {
    return c.json({ error: 'Database connection failed' }, 503);
  }
  
  return c.json({ 
    error: 'Internal server error',
    requestId: c.req.header('x-request-id') || 'unknown'
  }, 500);
});
```
