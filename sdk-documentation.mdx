// Real-time activity monitoring
class ActivityMonitor {
  private client: SimClient;
  private lastTimestamp: string | null = null;

  constructor(client: SimClient) {
    this.client = client;
  }

  async pollForNewActivity(address: string, callback: (activities: Activity[]) => void) {
    const activity = await this.client.evm.activity(address, {
      startTime: this.lastTimestamp || new Date(Date.now() - 3600000).toISOString(),
      limit: 100
    });

    if (activity.activities.length > 0) {
      const newActivities = this.lastTimestamp 
        ? activity.activities.filter(a => a.timestamp > this.lastTimestamp!)
        : activity.activities;

      if (newActivities.length > 0) {
        callback(newActivities);
        this.lastTimestamp = newActivities[0].timestamp;
      }
    }
  }

  startPolling(address: string, intervalMs: number = 30000) {
    return setInterval(() => {
      this.pollForNewActivity(address, (activities) => {
        console.log(`Found ${activities.length} new activities`);
      });
    }, intervalMs);
  }
}

const monitor = new ActivityMonitor(client);
const pollId = monitor.startPolling('0x...', 15000); // Poll every 15 seconds

### Error Handling

```typescript
import { SimApiError, RateLimitError, AuthenticationError } from '@sim/api-client';

try {
  const balances = await client.evm.balances('0x...');
} catch (error) {
  if (error instanceof RateLimitError) {
    console.log(`Rate limited. Retry after: ${error.retryAfter}ms`);
    // Wait and retry
    await new Promise(resolve => setTimeout(resolve, error.retryAfter));
    const balances = await client.evm.balances('0x...');
  } else if (error instanceof AuthenticationError) {
    console.error('Invalid API key');
  } else if (error instanceof SimApiError) {
    console.error(`API Error: ${error.message} (${error.statusCode})`);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

### Async Batch Processing

```python
import asyncio
from typing import List

class WalletAnalyzer:
    def __init__(self, client: SimClient):
        self.client = client
    
    async def analyze_wallet(self, address: str) -> dict:
        """Comprehensive wallet analysis"""
        tasks = [
            self.client.evm.balances(address),
            self.client.evm.activity(address, limit=100),
            self.client.evm.collectibles(address)
        ]
        
        balances, activity, nfts = await asyncio.gather(*tasks)
        
        return {
            'address': address,
            'total_value': sum(b.value_usd for b in balances.balances),
            'token_count': len(balances.balances),
            'activity_count': len(activity.activities),
            'nft_count': len(nfts.collectibles),
            'chains': list(set(b.chain for b in balances.balances))
        }
    
    async def analyze_multiple_wallets(self, addresses: List[str]) -> List[dict]:
        """Analyze multiple wallets concurrently"""
        tasks = [self.analyze_wallet(addr) for addr in addresses]
        return await asyncio.gather(*tasks)

# Usage
analyzer = WalletAnalyzer(client)
addresses = ['0x...', '0x...', '0x...']
results = asyncio.run(analyzer.analyze_multiple_wallets(addresses))

for result in results:
    print(f"Wallet {result['address']}: ${result['total_value']:.2f}")
```

### Error Handling

```python
from sim_api.exceptions import (
    SimApiError,
    RateLimitError,
    AuthenticationError,
    ValidationError
)
import time

def safe_api_call(func, *args, **kwargs):
    """Wrapper for safe API calls with retry logic"""
    max_retries = 3
    
    for attempt in range(max_retries):
        try:
            return func(*args, **kwargs)
            
        except RateLimitError as e:
            if attempt == max_retries - 1:
                raise
            
            print(f"Rate limited, waiting {e.retry_after}s...")
            time.sleep(e.retry_after / 1000)  # Convert ms to seconds
            
        except AuthenticationError:
            print("Authentication failed - check your API key")
            raise
            
        except ValidationError as e:
            print(f"Validation error: {e.message}")
            raise
            
        except SimApiError as e:
            print(f"API error: {e.message} (status: {e.status_code})")
            if attempt == max_retries - 1:
                raise
            
            time.sleep(2 ** attempt)  # Exponential backoff

# Usage
balances = safe_api_call(
    client.evm.balances,
    '0xd8da6bf26964af9d7eed9e03e53415d37aa96045'
)
```

func GetAllBalances(ctx context.Context, client *sim.Client, address string) ([]sim.Balance, error) {
    var allBalances []sim.Balance
    var offset *string
    
    for {
        opts := &sim.BalanceOptions{
            Limit:  100,
            Offset: offset,
        }
        
        response, err := client.EVM.Balances(ctx, address, opts)
        if err != nil {
            return nil, err
        }
        
        allBalances = append(allBalances, response.Balances...)
        
        if response.NextOffset == nil {
            break
        }
        
        offset = response.NextOffset
    }
    
    return allBalances, nil
}

### Concurrent Processing

```go
func AnalyzeMultipleWallets(ctx context.Context, client *sim.Client, addresses []string) ([]WalletAnalysis, error) {
    type result struct {
        analysis WalletAnalysis
        err      error
    }
    
    resultChan := make(chan result, len(addresses))
    
    // Start goroutines for each wallet
    for _, addr := range addresses {
        go func(address string) {
            analysis, err := AnalyzeWallet(ctx, client, address)
            resultChan <- result{analysis: analysis, err: err}
        }(addr)
    }
    
    // Collect results
    var analyses []WalletAnalysis
    var errors []error
    
    for i := 0; i < len(addresses); i++ {
        res := <-resultChan
        if res.err != nil {
            errors = append(errors, res.err)
        } else {
            analyses = append(analyses, res.analysis)
        }
    }
    
    if len(errors) > 0 {
        return analyses, fmt.Errorf("encountered %d errors", len(errors))
    }
    
    return analyses, nil
}

type WalletAnalysis struct {
    Address     string  `json:"address"`
    TotalValue  float64 `json:"total_value"`
    TokenCount  int     `json:"token_count"`
    ChainCount  int     `json:"chain_count"`
}

func AnalyzeWallet(ctx context.Context, client *sim.Client, address string) (WalletAnalysis, error) {
    balances, err := client.EVM.Balances(ctx, address, nil)
    if err != nil {
        return WalletAnalysis{}, err
    }
    
    var totalValue float64
    chains := make(map[string]bool)
    
    for _, balance := range balances.Balances {
        totalValue += balance.ValueUSD
        chains[balance.Chain] = true
    }
    
    return WalletAnalysis{
        Address:    address,
        TotalValue: totalValue,
        TokenCount: len(balances.Balances),
        ChainCount: len(chains),
    }, nil
}
```

## REST API Direct Integration

For languages without official SDKs, you can integrate directly with the REST API.

### cURL Examples

```bash
# Authentication header
API_KEY="your-api-key"
BASE_URL="https://api.sim.dune.com"

# Get balances
curl -X GET "$BASE_URL/v1/evm/balances/0x..." \
  -H "X-Sim-Api-Key: $API_KEY" \
  -H "Content-Type: application/json"

# Get activity with parameters
curl -X GET "$BASE_URL/v1/evm/activity/0x...?limit=50&chain_ids=ethereum" \
  -H "X-Sim-Api-Key: $API_KEY"

# Handle errors
curl -X GET "$BASE_URL/v1/evm/balances/invalid-address" \
  -H "X-Sim-Api-Key: $API_KEY" \
  -w "HTTP Status: %{http_code}\n" \
  -s -o response.json

# Pagination
OFFSET=""
while true; do
  RESPONSE=$(curl -s -X GET "$BASE_URL/v1/evm/balances/0x...?limit=100&offset=$OFFSET" \
    -H "X-Sim-Api-Key: $API_KEY")
  
  # Process response
  echo "$RESPONSE" | jq '.balances[]'
  
  # Get next offset
  OFFSET=$(echo "$RESPONSE" | jq -r '.next_offset // empty')
  
  # Break if no more data
  if [ -z "$OFFSET" ]; then
    break
  fi
done
```

        $this->apiKey = $apiKey;
        $this->baseUrl = $baseUrl;
        $this->httpClient = new \GuzzleHttp\Client([
            'timeout' => 30,
            'headers' => [
                'X-Sim-Api-Key' => $this->apiKey,
                'User-Agent' => 'SimAPI-PHP/1.0'
            ]
        ]);
    }
    
    public function getBalances($address, $options = []) {
        $url = "/v1/evm/balances/{$address}";
        
        if (!empty($options)) {
            $url .= '?' . http_build_query($options);
        }
        
        try {
            $response = $this->httpClient->get($this->baseUrl . $url);
            return json_decode($response->getBody(), true);
        } catch (\GuzzleHttp\Exception\ClientException $e) {
            if ($e->getResponse()->getStatusCode() === 429) {
                throw new Exception('Rate limit exceeded');
            }
            throw new Exception('API Error: ' . $e->getMessage());
        }
    }
    
    public function getAllBalances($address) {
        $allBalances = [];
        $offset = null;
        
        do {
            $options = ['limit' => 100];
            if ($offset) {
                $options['offset'] = $offset;
            }
            
            $response = $this->getBalances($address, $options);
            $allBalances = array_merge($allBalances, $response['balances']);
            $offset = $response['next_offset'] ?? null;
            
        } while ($offset);
        
        return $allBalances;
    }
}

## Best Practices

### Rate Limiting

All SDKs implement automatic rate limiting and retry logic:

```typescript
// JavaScript - Custom retry logic
const retryableRequest = async (fn, maxRetries = 3) => {
  for (let i = 0; i <= maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (error instanceof RateLimitError && i < maxRetries) {
        await new Promise(resolve => setTimeout(resolve, error.retryAfter));
        continue;
      }
      throw error;
    }
  }
};
```

### Caching

Implement caching to reduce API calls:

```python
# Python - Simple caching
from functools import lru_cache
from datetime import datetime, timedelta

class CachedSimClient:
    def __init__(self, client):
        self.client = client
        self._cache = {}
        self._cache_ttl = timedelta(minutes=5)
    
    def _is_cache_valid(self, timestamp):
        return datetime.now() - timestamp < self._cache_ttl
    
    def get_balances(self, address):
        cache_key = f"balances:{address}"
        
        if cache_key in self._cache:
            data, timestamp = self._cache[cache_key]
            if self._is_cache_valid(timestamp):
                return data
        
        data = self.client.evm.balances(address)
        self._cache[cache_key] = (data, datetime.now())
        return data
```

### Error Handling

Always implement proper error handling:

```go
// Go - Comprehensive error handling
func safeAPICall[T any](ctx context.Context, call func() (T, error)) (T, error) {
    var zero T
    
    backoff := time.Second
    maxBackoff := 30 * time.Second
    maxRetries := 3
    
    for attempt := 0; attempt <= maxRetries; attempt++ {
        result, err := call()
        if err == nil {
            return result, nil
        }
        
        // Check if we should retry
        if rateLimitErr, ok := err.(*errors.RateLimitError); ok {
            if attempt < maxRetries {
                sleepTime := time.Duration(rateLimitErr.RetryAfter) * time.Millisecond
                time.Sleep(sleepTime)
                continue
            }
        }
        
        // For other errors, use exponential backoff
        if attempt < maxRetries {
            time.Sleep(backoff)
            backoff *= 2
            if backoff > maxBackoff {
                backoff = maxBackoff
            }
            continue
        }
        
        return zero, err
    }
    
    return zero, fmt.Errorf("max retries exceeded")
}
```
